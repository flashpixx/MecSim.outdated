193  /**
194  * decodes the Jason parameter list to create the Java signature call in the structure
195  * "commandname(bind, method, [return type / default void | [return type, return belief name]], [method type arguments / default void], ...arguments)"
196  *
197  * @param p_parameter Jason parameter
198  * @param p_argumenttypes append list with method argument types
199  * @param p_argumentdata append list with method argument data
200  * @return returns a pair with class and belief name of the return paramater
201  *
202  * @throws ClassNotFoundException on class error
203  * @throws NoValueException on value error
204  */
205  private org.apache.commons.lang3.tuple.Pair<Class<?>, String> decodeMethodSignature( final List<Term> p_parameter, final List<Class<?>> p_argumenttypes,
206  final List<Object> p_argumentdata
207  ) throws ClassNotFoundException, NoValueException, IllegalAccessException
208  {
209  // class and string are immutable types, so we need to create a return argument
210  Class<?> l_returntype = void.class;
211  String l_returnname = CCommon.clearString( p_parameter.get( 1 ).toString() );
212 
213  // if paramter count > 2, then return type exists (single value or list value with 2 elements)
214  if ( p_parameter.size() > 2 )
215  if ( !p_parameter.get( 2 ).isList() )
216  l_returntype = this.convertTermToClass( p_parameter.get( 2 ) );
217  else
218  {
219  final ListTerm l_list = ( (ListTerm) p_parameter.get( 2 ) );
220  if ( l_list.size() != 2 )
221  throw new IllegalArgumentException( de.tu_clausthal.in.mec.common.CCommon.getResourceString( this, "returnargument" ) );
222 
223  l_returntype = this.convertTermToClass( l_list.get( 0 ) );
224  l_returnname = CCommon.clearString( l_list.get( 1 ).toString() );
225  }
226 
227  // if parameter count > 3, method arguments and argument types are exists
228  if ( p_parameter.size() > 3 )
229  {
230  if ( p_parameter.get( 3 ).isList() )
231  p_argumenttypes.addAll( Arrays.asList( this.convertTermListToClassArray( (ListTerm) p_parameter.get( 3 ) ) ) );
232  else
233  p_argumenttypes.add( this.convertTermToClass( p_parameter.get( 3 ) ) );
234 
235  if ( p_parameter.size() - 4 != p_argumenttypes.size() )
236  throw new IllegalArgumentException( de.tu_clausthal.in.mec.common.CCommon.getResourceString( this, "argumentnumber" ) );
237 
238  for ( int i = 4; i < p_parameter.size(); ++i )
239  p_argumentdata.add( CCommon.getJavaValue( p_parameter.get( i ), p_argumenttypes.get( i - 4 ) ) );
240  }
241 
242  // return immutable types
243  return new ImmutablePair<>( l_returntype, l_returnname );
244  }
